      # 创建 Web 项目

      # 1. 下载脚手架工程

      $ svn export https://192.168.1.118:1443/svn/KunLun/Client/kunlun-client-web-boilerplate

      # 2. 本地运行

      // 跳转到工程根目录
$ cd kunlun-client-web-boilerplate

// 安装依赖
$ yarn install

// 执行测试
$ yarn test

// 启动应用
$ yarn start

      # 3. 工程目录结构

      • ES6

      • 数据库：Dexie.js
      • 单元测试：Jest、Enzyme
